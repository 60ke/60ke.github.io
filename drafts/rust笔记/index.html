extern crate hex;
extern crate serde;
use std::collections::HashMap;
use serde::ser::{Serialize, Serializer, SerializeSeq, SerializeMap, SerializeStruct};
#[macro_use]
extern crate serde_derive;




type Dictionary = HashMap<String, String>;


#[derive(Debug,Serialize, Deserialize,Clone)]
pub struct Account {
    pub address: String,
    pub balance: usize,
    pub code: String,
    pub storage: Dictionary,
}

#[derive(Debug,Serialize, Deserialize,Clone)]
pub struct WriteSet {
    pub accounts: Vec<Account>,
    pub gas: usize,
}


impl Default for WriteSet {
    fn default() -> WriteSet {
        WriteSet{
            accounts: Vec::new(),
            gas: 0
        }
    }
}

impl Account{
    pub fn address(&self) -> String {
        self.address.clone()
    }
}


fn get_write_addrlist(writeset : & WriteSet) -> Vec<String>{
    // let mut v = Vec::new();
    let mut vec: Vec<String> = Vec::new();
    for account in &writeset.accounts{
        vec.push(account.address())
    }
    vec.dedup();
    vec
}



fn push_write_account(mut writeset : WriteSet,accounts:Vec<Account>) -> WriteSet{
for account in accounts{    
    let mut writeset_copy = writeset.clone();
    let mut writeset_accs = get_write_addrlist(&writeset);
    if ! writeset_accs.contains(&account.address()){
        writeset.accounts.push(account);
    }else{
        // let index = writeset.accounts.iter();
        let index = writeset.accounts.iter().position(|r| r.address == account.address()).unwrap();
        for (k,v) in account.storage{
            writeset.accounts[index].storage.insert(k, v);
        }
        // writeset.accounts[index] = account;
        println!("index {:?}", writeset.accounts[index]);
    }
}
    return writeset

}
