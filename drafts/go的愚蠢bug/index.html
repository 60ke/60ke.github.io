package main

import (
	"fmt"
	"os"

	toml "github.com/pelletier/go-toml"
)

type Config struct {
	Array []string `toml:"array" comment:"array test"`
}

func main() {
	var conf Config
	conf.Array = append(conf.Array, "aaa")
	conf.Array = append(conf.Array, "aab")
	cfile, err := os.Create("test.toml")
	if err != nil {
		fmt.Println(err)
	}
	defer cfile.Close()

	enc := toml.NewEncoder(cfile)
	enc.Order(toml.OrderPreserve)
	fmt.Println(conf)
	if err := enc.Encode(&conf); err != nil {
		fmt.Println(err)
	}
}
