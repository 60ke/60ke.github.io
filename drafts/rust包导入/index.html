pub type U256 = [u8; 32];


#[derive(Debug)]
pub struct Sink1{
    pub capacity:usize,
    pub preamble:Vec<u8>,
    pub heap:Vec<u8>
}

fn from(num:u32) -> U256{
    let mut slice = [0u8;32];
    slice[28] = (num>>24) as u8;
    slice[29] = (num>>16) as u8;
    slice[30] = (num>>8) as u8;
    slice[31] = num as u8;
    slice
}




impl From<u32> for Sink1 {
    fn from(num: u32) -> Self {
        let mut slice = [0u8;32];
        slice[28] = (num>>24) as u8;
        slice[29] = (num>>16) as u8;
        slice[30] = (num>>8) as u8;
        slice[31] = num as u8;
        Sink1{
            capacity:1 as usize,
            preamble:slice.into(),
            heap:Vec::new(),
        }
    }
}


trait Codec:Sized {
    fn encode(self,sink: &mut Sink);
}
impl Codec for u32{
    fn encode(self,sink: &mut Sink){
        let tar = from(self);
        sink.preamble.extend_from_slice(&tar);
    }
}
fn main() {
    let a:u32 = 8;
    let mut sink = Sink{
        capacity:1,
        preamble:Vec::new(),
        heap:Vec::new(),
    };
    a.encode(&mut sink);
    println!("{:?}",sink);

    let sink1 = Sink1::from(3);
