rust包导入

Feb 03, 2021
阅读时长: 3 分钟

rust 新版本导入包使用,可以不用先extern crate只需在cargo.toml声明,然后可直接用use

我们创建个项目codec_test

当rust中同时存在lib.rs,main.rs时无法使用,use crate::types::* 错误提示: help: a similar path exists: codec_test::types,这是因为在lib.rs中对于lib.rs来说crate::代表了codec_testcrate本身,而对于main.rs来说则相当于是mian.rs导入了名为codec_test的外部crate

use std::convert::From;

// use crate::types::Sink;
use codec_test::types::Sink;

// use codec_test::types::Sink1;

pub type U256 = [u8; 32];


#[derive(Debug)]
pub struct Sink1{
    pub capacity:usize,
    pub preamble:Vec<u8>,
    pub heap:Vec<u8>
}

fn from(num:u32) -> U256{
    let mut slice = [0u8;32];
    slice[28] = (num>>24) as u8;
    slice[29] = (num>>16) as u8;
    slice[30] = (num>>8) as u8;
    slice[31] = num as u8;
    slice
}




impl From<u32> for Sink1 {
    fn from(num: u32) -> Self {
        let mut slice = [0u8;32];
        slice[28] = (num>>24) as u8;
        slice[29] = (num>>16) as u8;
        slice[30] = (num>>8) as u8;
        slice[31] = num as u8;
        Sink1{
            capacity:1 as usize,
            preamble:slice.into(),
            heap:Vec::new(),
        }
    }
}


trait Codec:Sized {
    fn encode(self,sink: &mut Sink);
}
impl Codec for u32{
    fn encode(self,sink: &mut Sink){
        let tar = from(self);
        sink.preamble.extend_from_slice(&tar);
    }
}
fn main() {
    let a:u32 = 8;
    let mut sink = Sink{
        capacity:1,
        preamble:Vec::new(),
        heap:Vec::new(),
    };
    a.encode(&mut sink);
    println!("{:?}",sink);

    let sink1 = Sink1::from(3);
    println!("{:?}",sink1);
    let sink2:Sink1 = 4.into();
    println!("sink2 {:?}",sink2); 
}
